			function Counter() {
				const [count, setCount] = useState(0);
				return (
					<div>
						<p>counter: {count}</p>
						<button onClick={() => setCount(count + 1)}>bigger</button>
						<button onClick={() => setCount(count - 1)}>smaller</button>
					</div>
				);
			}
			export default Counter;
